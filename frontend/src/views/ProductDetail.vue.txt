<template>
  <div class="product-detail">
    <!-- 商品轮播图 -->
    <div class="swiper-container">
      <van-swipe class="swiper" :autoplay="3000" indicator-color="white">
        <van-swipe-item v-for="(img, index) in product.images" :key="index">
          <img :src="img" class="swiper-img" />
        </van-swipe-item>
      </van-swipe>
    </div>

    <!-- 商品信息卡片 -->
    <div class="info-card">
      <!-- 商品标题 -->
      <h1 class="title">{{ product.title }}</h1>
      
      <!-- 价格 -->
      <div class="price-box">
        <span class="price-label">¥</span>
        <span class="price">{{ product.price }}</span>
      </div>

      <!-- 基础信息标签 -->
      <div class="meta-info">
        <span class="tag">{{ formatCondition(product.conditionLevel) }}</span>
        <span class="tag">{{ product.campus }}校区</span>
        <span class="tag">浏览 {{ product.viewCount || 100 }} 次</span>
      </div>
    </div>

    <!-- 卖家信息栏 -->
    <div class="seller-card" @click="goToSellerPage">
      <div class="seller-avatar">
        <van-image
          round
          width="40px"
          height="40px"
          :src="product.sellerAvatar || 'https://via.placeholder.com/40'"
          fit="cover"
        />
      </div>
      <div class="seller-info">
        <div class="seller-name">{{ product.sellerName || '校园闲置小助手' }}</div>
        <div class="seller-stats">
          <!-- 这里假设 sellerRating 是一个数字，比如 4.8 -->
          <van-rate v-model="product.sellerRating" readonly size="12" color="#ff9800" />
          <span class="rating-text">{{ product.sellerRating }}分</span>
          <span class="sep">|</span>
          <span>成交 {{ product.sellerDealCount || 10 }} 单</span>
        </div>
      </div>
      <div class="seller-action">
        <van-button size="small" type="primary" @click.stop="contactSeller">联系卖家</van-button>
      </div>
    </div>

    <!-- 评价/交易记录区域 -->
    <div class="comment-section">

      </div>

        <!-- 如果有评价数据，遍历显示 -->
        <div v-if="product.reviews && product.reviews.length > 0">
          <div class="comment-item" v-for="(review, index) in product.reviews" :key="index">
            <div class="comment-user">
              <van-image round width="24px" height="24px" :src="review.avatar" />
              <span>{{ review.userName }}</span>
            </div>
            <div class="comment-content">{{ review.content }}</div>
            <div class="comment-time">{{ review.timeAgo }}</div>
          </div>

        <!-- 如果没有评价，显示默认提示 -->
        <div v-else class="no-comments">

      <!-- 这里可以放更长的描述，或者原图 -->
      <p v-if="product.description">{{ product.description }}</p>
      <p v-else>暂无更多描述。</p>
    </div>

    <!-- 底部操作栏 -->
    <div class="detail-footer">
      <div class="footer-actions">
        <div class="action-item" @click="toggleFavorite">
          <van-icon :name="isFavorite ? 'like' : 'like-o'" size="20" :color="isFavorite ? '#ee0a24' : ''" />
          <span :style="{ color: isFavorite ? '#ee0a24' : '' }">{{ isFavorite ? '已收藏' : '收藏' }}</span>
        </div>

      <div class="footer-buttons">
        <van-button type="warning" size="small" @click="addToCart">加入购物车</van-button>

      </div>
    </div>

<script>
import axios from 'axios'

export default {
  name: 'ProductDetail',
    data() {
    return {
      product: {
        id: '',
        title: '',
        price: 0,
        conditionLevel: '',
        campus: '',
        categoryName: '',
        images: [], // 轮播图数组
        description: '',
        sellerName: '',
        sellerAvatar: '',
        sellerRating: 5.0, // 卖家评分
        sellerDealCount: 0, // 卖家成交数
        viewCount: 0, // 浏览量
        reviews: [] // 评价列表
      },
      isFavorite: false // 是否已收藏
    }
  },
  async created() {
    await this.fetchProductDetail()
  },
  methods: {
    // 获取商品详情
    async fetchProductDetail() {
      const id = this.$route.params.id
      try {
        const res = await axios.get(`/api/products/${id}`)
        
        // 假设后端返回的数据结构在 data 字段里，根据实际情况调整
        const data = res.data.data || res.data

        // 格式化价格
        const formattedPrice = Number(data.price).toFixed(2)

        // 处理图片：如果 image 是单个字符串，转成数组
        let imageList = []
        if (Array.isArray(data.images)) {
          imageList = data.images
        } else if (data.image) {
          imageList = [data.image]
        }

        // 处理评价数据
        const reviews = data.reviews || []

        this.product = {
          ...this.product,
          id: data.id,
          title: data.title,
          price: formattedPrice,
          conditionLevel: data.conditionLevel,
          campus: data.campus,
          categoryName: data.categoryName,
          images: imageList,
          description: data.description,
          sellerName: data.sellerName || '匿名用户',
          sellerAvatar: data.sellerAvatar,
          sellerRating: data.sellerRating || 5.0,
          sellerDealCount: data.sellerDealCount || 0,
          viewCount: data.viewCount || 0,
          reviews: reviews
        }
      } catch (e) {
        console.error('获取商品详情失败', e)
        Toast('商品加载失败')
      }
    },

    // 联系卖家
    contactSeller() {
      Toast('正在跳转至聊天界面...')
      // 这里可以跳转到聊天页面
    },

    // 收藏商品
    toggleFavorite() {
      this.isFavorite = !this.isFavorite
      Toast(this.isFavorite ? '已收藏' : '已取消收藏')
    },

    // 加入购物车
    addToCart() {
      Toast('已加入购物车')
      // 这里可以调用 API 添加到购物车
    },

    // 立即购买
    buyNow() {
      Toast('正在创建订单...')
      // 这里可以跳转到订单确认页
    },

    // 跳转购物车
    goToCart() {
      this.$router.push('/cart')
    },

    // 跳转卖家主页
    goToSellerPage() {
      // Toast('跳转卖家主页')
    },

    // 格式化成色
    formatCondition(level) {
      const map = { 1: '全新', 2: '99新', 3: '95新', 4: '9成新', 5: '8成新', 6: '7成新及以下' }
      return map[level] || '未知成色'
    }
  }
}
</script>
